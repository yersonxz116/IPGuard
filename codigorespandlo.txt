static const char* STREAM_BOUNDARY =

  html += "}";

  html += "</script>";

  html += "</body>";
  html += "</html>";

  server.send(200, "text/html", html);
}

// =======================
// Encender LED
// =======================
void handleLedOn() {
  digitalWrite(LED_BLANCO, HIGH);
  ledEncendido = true;

  Serial.println("LED encendido");

  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Cache-Control", "no-store");
  server.send(200, "text/plain", "LED encendido");
}

// =======================
// Apagar LED
// =======================
void handleLedOff() {
  digitalWrite(LED_BLANCO, LOW);
  ledEncendido = false;

  Serial.println("LED apagado");

  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Cache-Control", "no-store");
  server.send(200, "text/plain", "LED apagado");
}

// =======================
// Estado LED
// =======================
void handleLedStatus() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  server.sendHeader("Cache-Control", "no-store");

  if (ledEncendido) {
    server.send(200, "text/plain", "on");
  } else {
    server.send(200, "text/plain", "off");
  }
}

// =======================
// Streaming MJPEG baja latencia
// =======================
void handleStream() {
  WiFiClient client = server.client();

  Serial.println("Cliente conectado al streaming");

  client.setNoDelay(true);

  String response = "HTTP/1.1 200 OK\r\n";
  response += "Content-Type: ";
  response += STREAM_CONTENT_TYPE;
  response += "\r\n";
  response += "Access-Control-Allow-Origin: *\r\n";
  response += "Cache-Control: no-store, no-cache, must-revalidate, max-age=0\r\n";
  response += "Pragma: no-cache\r\n";
  response += "Expires: 0\r\n";
  response += "Connection: close\r\n\r\n";

  client.print(response);

  while (client.connected()) {
    camera_fb_t *fb = esp_camera_fb_get();

    if (!fb) {
      Serial.println("Error capturando frame");
      delay(1);
      continue;
    }

    client.print(STREAM_BOUNDARY);

    char partBuf[96];
    int hlen = snprintf(
      partBuf,
      sizeof(partBuf),
      "Content-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n",
      fb->len
    );

    client.write((const uint8_t*)partBuf, hlen);

    size_t written = client.write(fb->buf, fb->len);

    esp_camera_fb_return(fb);

    if (written == 0) {
      Serial.println("Cliente desconectado o lento");
      break;
    }

    delay(1);
  }

  Serial.println("Cliente desconectado del streaming");
}

// =======================
// Foto individual opcional
// =======================
void handleFoto() {
  camera_fb_t *fb = esp_camera_fb_get();

  if (!fb) {
    Serial.println("Error tomando foto");
    server.send(500, "text/plain", "Error tomando foto");
    return;
  }

  WiFiClient client = server.client();

  server.setContentLength(fb->len);
  server.send(200, "image/jpeg", "");

  client.write(fb->buf, fb->len);

  esp_camera_fb_return(fb);
}

// =======================
// Iniciar camara
// =======================
bool iniciarCamara() {
  camera_config_t config;

  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;

  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;

  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;

  config.pin_sccb_sda = SIOD_GPIO_NUM;
  config.pin_sccb_scl = SIOC_GPIO_NUM;

  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;

  // =======================
  // Configuracion para baja latencia
  // =======================
  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;

  // Mejor para tiempo real:
  // FRAMESIZE_QQVGA = menor delay
  // FRAMESIZE_QVGA  = mejor calidad, mas delay
  config.frame_size = FRAMESIZE_QQVGA;

  // Menor numero = mas calidad y mas peso
  // Mayor numero = menos calidad y menos delay
  config.jpeg_quality = 22;

  if (psramFound()) {
    config.fb_count = 2;
    config.fb_location = CAMERA_FB_IN_PSRAM;
    config.grab_mode = CAMERA_GRAB_LATEST;
    Serial.println("PSRAM detectada: usando modo baja latencia");
  } else {
    config.fb_count = 1;
    config.fb_location = CAMERA_FB_IN_DRAM;
    config.grab_mode = CAMERA_GRAB_LATEST;
    Serial.println("PSRAM no detectada: usando DRAM");
  }

  Serial.println("Iniciando camara...");

  esp_err_t err = esp_camera_init(&config);

  if (err != ESP_OK) {
    Serial.print("Error iniciando camara: 0x");
    Serial.println(err, HEX);
    return false;
  }

  Serial.println("Camara inicializada correctamente");

  sensor_t *s = esp_camera_sensor_get();

  if (s != NULL) {
    s->set_framesize(s, FRAMESIZE_QQVGA);
    s->set_quality(s, 22);

    s->set_brightness(s, 0);
    s->set_contrast(s, 0);
    s->set_saturation(s, 0);

    s->set_gain_ctrl(s, 1);
    s->set_exposure_ctrl(s, 1);
    s->set_awb_gain(s, 1);

    s->set_special_effect(s, 0);
    s->set_whitebal(s, 1);
  }

  return true;
}

// =======================
// Setup
// =======================
void setup() {
  Serial.begin(115200);
  delay(2000);

  pinMode(LED_BLANCO, OUTPUT);
  digitalWrite(LED_BLANCO, LOW);
  ledEncendido = false;

  Serial.println();
  Serial.println("Iniciando ESP32-CAM Streaming Realtime...");

  if (!iniciarCamara()) {
    Serial.println("No se pudo iniciar la camara");
    return;
  }

  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);

  // Importante para reducir latencia WiFi
  WiFi.setSleep(false);

  Serial.print("Conectando a WiFi");

  int intentos = 0;

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
    intentos++;

    if (intentos > 60) {
      Serial.println();
      Serial.println("No se pudo conectar al WiFi");
      Serial.println("Revisa nombre, clave o que sea red 2.4GHz");
      return;
    }
  }

  Serial.println();
  Serial.println("WiFi conectado");
  Serial.print("IP: ");
  Serial.println(WiFi.localIP());

  Serial.print("Abre en el navegador: http://");
  Serial.println(WiFi.localIP());

  server.on("/", HTTP_GET, handleRoot);
  server.on("/stream", HTTP_GET, handleStream);
  server.on("/foto", HTTP_GET, handleFoto);

  server.on("/led/on", HTTP_GET, handleLedOn);
  server.on("/led/off", HTTP_GET, handleLedOff);
  server.on("/led/status", HTTP_GET, handleLedStatus);

  server.begin();

  Serial.println("Servidor iniciado");
}

// =======================
// Loop
// =======================
void loop() {
  server.handleClient();
}
